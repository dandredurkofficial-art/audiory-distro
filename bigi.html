import express from 'express'; import cors from 'cors'; import dotenv from 'dotenv'; import mongoose from 'mongoose'; dotenv.config(); const app = express(); const PORT = process.env.PORT || 5000; // Request Middlewares app.use(cors()); app.use(express.json()); // MongoDB Database Connection Hook mongoose.connect(process.env.MONGO_URI) .then(() => console.log('📁 Database connected successfully to MongoDB Atlas!')) .catch((err) => console.error('❌ Database connection error details:', err)); // Test Connection Route app.get('/api/test', (req, res) => { res.json({ message: "Audiory Distro API is live and running!" }); }); app.listen(PORT, () => { console.log(`🚀 Server is flying high on port ${PORT}`); });
