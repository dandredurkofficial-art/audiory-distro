import express from 'express'; import cors from 'cors'; import dotenv from 'dotenv'; dotenv.config(); const app = express(); const PORT = process.env.PORT || 5000; // Request Middlewares app.use(cors()); app.use(express.json()); // Test Connection Route app.get('/api/test', (req, res) => { res.json({ message: "Audiory Distro API is live and running!" }); }); app.listen(PORT, () => { console.log(`🚀 Server is flying high on port ${PORT}`); });
