import express from 'express'; import cors from 'cors'; import path from 'path'; import fs from 'fs'; import multer from 'multer'; import admin from 'firebase-admin'; import { fileURLToPath } from 'url'; // Reconstruct __dirname for ES Modules compatibility const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); // 1. Initialize Firebase Admin using modern ES Module 'with' syntax import serviceAccount from './firebase-key.json' with { type: 'json' }; admin.initializeApp({ credential: admin.credential.cert(serviceAccount) }); const db = admin.firestore(); const app = express(); app.use(cors()); app.use(express.json()); app.use(express.static(path.join(__dirname, 'public'))); // 2. Configure Local Multi-File Storage Subsystems const uploadDir = path.join(__dirname, 'public', 'uploads'); if (!fs.existsSync(uploadDir)) { fs.mkdirSync(uploadDir, { recursive: true }); } const storage = multer.diskStorage({ destination: (req, file, cb) => { cb(null, uploadDir); }, filename: (req, file, cb) => { const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9); cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname)); } }); const upload = multer({ storage: storage }); // Define multi-field receiver configuration maps const distributionFields = [ { name: 'coverArt', maxCount: 1 }, { name: 'audioTracks', maxCount: 20 } ]; // 3. API ENDPOINTS // GET: Fetch Active Dynamic Production Catalog app.get('/api/releases', async (req, res) => { try { const snapshot = await db.collection('releases').orderBy('createdAt', 'desc').get(); const releases = []; snapshot.forEach(doc => { releases.push({ id: doc.id, ...doc.data() }); }); res.json(releases); } catch (error) { console.error('Error reading catalog snapshot:', error); res.status(500).json({ error: 'Failed to extract active release states.' }); } }); // POST: Advanced Multi-File Multi-Step Form Processor app.post('/api/releases', upload.fields(distributionFields), async (req, res) => { try { // Structural metadata array parsing from multi-part data payload string bindings const bodyData = req.body.jsonData ? JSON.parse(req.body.jsonData) : req.body; const { title, version, type, primaryArtist, featuredArtists, producerName, songwriters, composers, genre, subgenre, language, isExplicit, releaseDate, originalReleaseDate, recordLabel, copyrightOwner, publishingOwner, bpm, musicalKey, mood, tags, ISRC, splits, paymentMethod, mpesaNumber, paypalEmail, platforms, socialMonetization, tracksData } = bodyData; if (!title || !type || !primaryArtist) { return res.status(400).json({ error: 'Core properties missing: Title, Type, and Primary Artist are required.' }); } // Process uploaded file pointer locations let coverArtPath = ''; if (req.files && req.files['coverArt']) { coverArtPath = `/uploads/${req.files['coverArt'][0].filename}`; } let trackFiles = []; if (req.files && req.files['audioTracks']) { trackFiles = req.files['audioTracks'].map(f => `/uploads/${f.filename}`); } // Combine tracks mapping data fields with physical binary paths const finalTracksArray = (tracksData || []).map((track, index) => ({ ...track, audioFileUrl: trackFiles[index] || '' })); const compiledReleasePayload = { title, type, coverArtUrl: coverArtPath, upc: bodyData.upc || `AUC-${Math.floor(10000000 + Math.random() * 90000000)}`, releaseDate: releaseDate || new Date().toLocaleDateString('en-US', { month: 'short', day: '2-digit', year: 'numeric' }), streams: bodyData.streams ? Number(bodyData.streams) : 0, status: 'In Review', createdAt: new Date().toISOString(), tracks: finalTracksArray, metadata: { version: version || 'Original Mix', primaryArtist, featuredArtists: featuredArtists || '', producerName: producerName || '', songwriters: songwriters || '', composers: composers || '', genre: genre || 'Urban / Local', subgenre: subgenre || '', language: language || 'Swahili', isExplicit: isExplicit === 'Yes' || isExplicit === true, originalReleaseDate: originalReleaseDate || releaseDate, recordLabel: recordLabel || 'Independent / Audiory', copyrightOwner: copyrightOwner || primaryArtist, publishingOwner: publishingOwner || primaryArtist }, technicalSpecs: { bpm: bpm || '', musicalKey: musicalKey || '', mood: mood || '', tags: tags || '', isrc: ISRC || `KE-AUD-${Math.floor(100000 + Math.random() * 900000)}` }, royaltyArchitecture: { splits: splits || [], paymentMethod: paymentMethod || 'M-Pesa', mpesaNumber: mpesaNumber || '', paypalEmail: paypalEmail || '' }, distributionTargets: { platforms: platforms || [], socialMonetization: socialMonetization || [] } }; const docRef = await db.collection('releases').add(compiledReleasePayload); res.status(201).json({ id: docRef.id, ...compiledReleasePayload }); } catch (error) { console.error('❌ Error handling multi-part pipeline submission:', error); res.status(500).json({ error: 'Failed to deploy multi-file distribution package.' }); } }); // Fallback logic delivering index framework routing app.get('*', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); const PORT = process.env.PORT || 5000; app.listen(PORT, () => { console.log(`🚀 Audiory Core Server humming nicely on port ${PORT}`); });
